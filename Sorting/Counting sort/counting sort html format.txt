#include&lt;iostream&gt;

using namespace std;

int main()
{
    long long int Data[99999]={0};
    long long int Element_number;
    long long int Max_element=0;
    long long int Element;

    cout&lt;&lt;"Enter the number of element : "&lt;&lt;endl;
    cin&gt;&gt;Element_number;
    cout&lt;&lt;"Enter Elements : ";

    for(int Counter=0; Counter&lt;Element_number; Counter++)
    {
        cin&gt;&gt;Element;
        Data[Element]++;
        Max_element=max(Max_element,Element); //This will detect the largest element of the data
    }

    /**Here we will print the Data in a sequence otherwise
    we have to do nothing **/

    cout&lt;&lt;"Sorted Elements : ";

    for(int Index=0; Index&lt;=Max_element; Index++)
    {
        for(int Counter=0; Counter&lt;Data[Index]; Counter++)
            cout&lt;&lt;Index&lt;&lt;" ";
    }
    cout&lt;&lt;endl;

    return 0;
}

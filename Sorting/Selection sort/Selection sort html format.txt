#include&lt;iostream&gt;
using namespace std;
void interchange(int *first_element, int *second_element)
{
    int temporary_variable;
    temporary_variable = *first_element;
    *first_element = *second_element;
    *second_element = temporary_variable;
}
void selection_sort(int data_list[], int element_number)
{
    int minimum_element_index;
    for(int step_counter = 0; step_counter &lt; element_number-1; step_counter++)
    {
        minimum_element_index = step_counter;
        for(int index = step_counter+1; index &lt; element_number; index++)
        {
            if(data_list[minimum_element_index] &gt; data_list[index])
                minimum_element_index = index;
        }
        if(step_counter != minimum_element_index)
            interchange( &data_list[step_counter], &data_list[minimum_element_index]);
    }
}
int main()
{
    int element_number;

    cout &lt;&lt; "Enter number of elements : " ;
    cin &gt;&gt; element_number;

    int data_list[element_number];

    cout &lt;&lt; "Enter your elements : " &lt;&lt; endl;
    for(int index = 0; index &lt; element_number ; index++)
    {
        cin &gt;&gt; data_list[index];
    }

    selection_sort(data_list, element_number);

    cout &lt;&lt; "Sorted data are : ";
    for(int index = 0; index &lt; element_number ; index++)
    {
        cout &lt;&lt; data_list[index] &lt;&lt; " ";
    }
    cout &lt;&lt; endl;
    return 0;
}

#include&lt;cstdio&gt;
#include&lt;iostream&gt;

using namespace std;

void bubble_sort(int Array[], int Elements_number)
{
    for(int line_counter = 0; line_counter &lt;Elements_number - 1; line_counter++)
    {
        for(int index = 0; index &lt;Elements_number-line_counter-1; index++)
        {
            if(Array[index] &gt; Array[index+1])
            {

                int temporary_variable = Array[index];
                Array[index] = Array[index+1];
                Array[index+1] = temporary_variable;
            }
        }
    }
}


int main()
{
    int Elements_number;

    cout &lt;&lt;;"Enter Elements number : ";
    cin &gt;&gt; Elements_number;

    int Array[Elements_number];

    cout &lt;&lt;;"Enter Elements: " &lt;&lt;;endl;
    for(int index = 0; index &lt;Elements_number ; index++)
        cin &gt;&gt; Array[index];

    bubble_sort(Array, Elements_number);

    cout &lt;&lt;;"Sorted Elements: ";
    for(int index = 0; index &lt;Elements_number ; index++)
        cout &lt;&lt;;Array[index] &lt;&lt;;" ";
    cout &lt;&lt;;endl;



    return 0;
}

#include&lt;iostream&gt;
using namespace std;

void insertion_sort(int data_list[], int element_number)
{
    for(int item_index = 1; item_index &lt; element_number; item_index++)
    {
        int item = data_list[item_index];
        int index;
        for( index = item_index-1; index &gt;= 0 && item &lt; data_list[index]; index--)
        {
            data_list[index+1] = data_list[index];
        }
        data_list[index+1] = item;
    }
}
int main()
{
    int element_number;

    cout &lt;&lt; "Enter number of elements : " ;
    cin &gt;&gt; element_number;

    int data_list[element_number];

    cout &lt;&lt; "Enter your elements : " &lt;&lt; endl;
    for(int index = 0; index &lt; element_number ; index++)
    {
        cin &gt;&gt; data_list[index];
    }

    insertion_sort(data_list, element_number);

    cout &lt;&lt; "Sorted data are : ";
    for(int index = 0; index &lt; element_number ; index++)
    {
        cout &lt;&lt; data_list[index] &lt;&lt; " ";
    }
    cout &lt;&lt; endl;
    return 0;
}
